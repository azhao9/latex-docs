\question[20] Let $G$ be a group of order $6$.\vspace{.05in}\noaddpoints
\begin{parts}

\part[5] How many $3$-Sylow subgroups are there in $G$?

\part[5] Show that $G$ contains at least one subgroup of order $2$.\vspace{.1in}

\noindent Assume, for the remaining part of the exercise, that $G$ is not cyclic.\vspace{.05in}

\part[5] Let $H$ be a subgroup of $G$ of order $2$. Consider the set $\Omega = \{aH|a\in G\}$
  of left cosets of
$H$ in $G$. $G$ acts on $\Omega$ as follows:
    \[
    G\times \Omega \to \Omega,\qquad (g,aH) \mapsto gaH,\qquad\forall g\in G,~\forall a\in G.
    \]
Determine the cardinality $|\Omega|$ of
$\Omega$.
\part[5] Let
    \[
    \varphi: G \to S_{|\Omega|}, \qquad \varphi(g)(aH) = gaH
    \]
    be the group homomorphism of $G$ into the group of permutations of $\Omega$. Determine $\text{Ker}(\varphi)$.
    \end{parts}    
\addpoints

Let $A=C([0,1],\mathbb R)$ be the ring of continuous (for the Euclidean topology) functions  $f:[0,1]\to\mathbb R$ and let $I\subset A$ be the subset of functions $f\in A$ such that $f(1/2)=0$.
    \begin{parts}
    \part[5] Show that $I$ an ideal of $A$.
    \part[5] Is $I$ a prime ideal? Prove it or disprove it.
    \part[10] Is $I$ a maximal ideal? Prove it or disprove it.
    \end{parts}
    \addpoints

\question[20]  Consider the polynomial $f(X)=X^2 +2X +3$ in $\mathbb{Z}_5[X]$.\noaddpoints
\begin{parts}
\part[5] Is $f(X)$ irreducible in $\mathbb{Z}_5[X]$? If yes, prove it, if not determine a proper factorization of $f(X)$ in $\mathbb{Z}_5[X]$.
\part[10] Let $I = (f(X))$ be the principal ideal in $\mathbb{Z}_5[X]$ generated by $f(X)$. Consider the factor ring $F=\mathbb{Z}_5[X]/I$.

Prove that the coset $\overline{X}:= X + I$ is invertible in $F$ (i.e.~find its multiplicative inverse) and determine
the order of $\overline{X}$ in the multiplicative group $F^{\times}$ of units of $F$.
\part[5] Find, if  exists, a coset of order 3 in $F^{\times}$.
\end{parts}
\addpoints
